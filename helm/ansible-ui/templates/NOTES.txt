{{- else if eq .Values.service.type "NodePort" }}
  Get the NodePort address:

    export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "ansible-ui.fullname" . }})
    export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
    echo "Open: http://$NODE_IP:$NODE_PORT"

{{- else if eq .Values.service.type "LoadBalancer" }}
  Wait for the LoadBalancer IP:

    kubectl get svc --namespace {{ .Release.Namespace }} {{ include "ansible-ui.fullname" . }} --watch

{{- else }}
  Forward the service port to your local machine:

    kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "ansible-ui.fullname" . }} 8080:{{ .Values.service.port }}

  Then open: http://localhost:8080

{{- end }}
